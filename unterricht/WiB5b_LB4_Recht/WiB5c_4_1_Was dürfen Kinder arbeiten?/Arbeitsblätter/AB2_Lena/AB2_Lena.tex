	\begin{infobox}
		\textit{„Hallo Rechtsdetektiv!\\
			Lena ist 14 Jahre alt und möchte in den Sommerferien im Café am Markt arbeiten. Die Chefin bietet an:}
		
		\begin{itemize}[leftmargin=*]
			\item \textbf{Arbeit:} Kuchen servieren und Tische abräumen
			\item \textbf{Zeit:} Montag bis Freitag von 10 bis 14 Uhr
			\item \textbf{Bezahlung:} 7 Euro pro Stunde
		\end{itemize}
		
		\textit{Deine Aufgabe: Überprüfe mit dem Gesetzes-Check, ob Lena diesen Job annehmen darf.}
	\end{infobox}
